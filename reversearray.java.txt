// Problem: Reverse an Array
// Approach: Two pointers
// Time Complexity: O(n)
// Space Complexity: O(1)

public class ReverseArray {

    public static void reverse(int[] arr) {
        int left = 0;
        int right = arr.length - 1;

        while (left < right) {
            int temp = arr[left];
            arr[left] = arr[right];
            arr[right] = temp;

            left++;
            right--;
        }
    }
}
